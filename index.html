        const itineraryContent = document.getElementById('itinerary-content');

        const dayButtons = document.querySelectorAll('.day-button');


        function renderDay(dayKey) {

            const dayData = itineraryData[dayKey];

            if (!dayData) {

                itineraryContent.innerHTML = '<p class="text-center text-red-500">No se encontró información para este día.</p>';

                return;

            }


            let html = `

                <div class="bg-white p-6 rounded-xl shadow-lg">

                    <h2 class="text-3xl font-bold text-blue-700 mb-4">${dayData.title}</h2>

                    <a href="${dayData.mapLink}" target="_blank" class="map-link">Ver Ruta en Google Maps</a>

                    <div class="mt-6 space-y-4">

            `;


            dayData.activities.forEach(activity => {

                html += `

                    <div class="activity-card">

                        <img src="${activity.imageUrl}" alt="${activity.name}" class="activity-image" onerror="this.onerror=null;this.src='https://placehold.co/400x180/CCCCCC/000?text=Imagen+no+disponible';">

                        <h3 class="text-xl font-semibold text-gray-900 mb-2">${activity.time}: ${activity.name}</h3>

                        <p class="text-gray-700 mb-2">${activity.description}</p>

                `;

                if (activity.links && activity.links.length > 0) {

                    html += `<div class="flex flex-wrap gap-2 mt-2">`;

                    activity.links.forEach(link => {

                        html += `<a href="${link.url}" target="_blank" class="text-blue-600 hover:underline text-sm">${link.text}</a>`;

                    });

                    html += `</div>`;

                }

                html += `</div>`; // Close activity-card

            });


            html += `

                    </div>

                </div>

            `;

            itineraryContent.innerHTML = html;


            // Update active button state

            dayButtons.forEach(button => button.classList.remove('active'));

            document.querySelector(`.day-button[data-day="${dayKey}"]`).classList.add('active');

        }


        // Event listeners for day buttons

        dayButtons.forEach(button => {

            button.addEventListener('click', () => {

                renderDay(button.dataset.day);

            });

        });


        // Render Day 1 by default on load

        document.addEventListener('DOMContentLoaded', () => {

            renderDay('day1');

        });

    </script>
